# PoE2 Trade API Research (Poe2Builds workspace)

## Key Finding: status: securable
- `status: "online"` = sellers currently in-game. Small pool, inflated prices. NOT useful for real prices.
- `status: "securable"` = instant-buy market via in-game trade system. This is the real competitive market price. ALWAYS use this.
- `status: "any"` = includes offline/spam listings, 10k+ garbage results. Avoid.

## Working API Pattern
POST `https://www.pathofexile.com/api/trade2/search/poe2/{leagueSlug}`
Headers: `Content-Type: application/json`, `X-Requested-With: XMLHttpRequest`
Body:
```json
{
  "query": {
    "status": { "option": "securable" },
    "name": "Item Name",
    "type": "Base Type",
    "stats": [{ "type": "and", "filters": [], "disabled": false }]
  },
  "sort": { "price": "asc" }
}
```
Returns: `{ id, total, result: [id1, id2, ...] }`

## Fetch Listings
GET `https://www.pathofexile.com/api/trade2/fetch/{id1,id2,...}?query={searchId}&realm=poe2`
- Max ~10 IDs per batch (safe limit)
- Wait 800-1000ms between batches to avoid rate limiting
- Returns: `{ result: [{ item: { name, typeLine }, listing: { price: { amount, currency } } }] }`

## Authentication
- Site uses Cloudflare + GGG session cookies (HttpOnly, not readable via document.cookie)
- API returns 403 from PowerShell/curl without auth
- SOLUTION: Use `run_playwright_code` with `page.evaluate(async () => { ... fetch(...) ... })`
  This runs fetch INSIDE the authenticated browser context — cookies attach automatically
- User must be logged in via the VS Code browser first (one-time per session)

## Price Calculation Strategy
- Fetch top 100 listings sorted by price asc
- Group by currency
- Compute median per currency
- Divine is the primary currency for expensive items
- Watch for price-fixing clusters (e.g., 99/100 results at same price — that IS the market price in PoE2's securable system)

## Price Tracking Page
- File: `e:\Code\Poe2Builds\prices.html` — self-contained, open with file:// directly
- Prices are embedded in `<script id="prices-data" type="application/json">` in the file
- To update: run the fetch script below via run_playwright_code, then update the embedded JSON
- Data structure: categories → array of bases → brackets with ilvlMin/ilvlMax and price stats
- Prices stored in Divine Orbs (floats). null = no data. count = total listings found.
- divineRateChaos: 1 divine = X chaos (used for display formatting only)

## Workflow for Price Updates

2. Navigate browser to trade URL to confirm session is active
3. Run the fetch script below via run_playwright_code (takes ~3-4 min total)
4. Script returns JSON with all price data
5. Update the <script id="prices-data"> block in prices.html with new data
6. Update lastUpdated timestamp and divineRateChaos if known

## Fetch Script (run via run_playwright_code → page.evaluate)
```javascript
return page.evaluate(async () => {
  const LEAGUE   = 'vaal';       // update when league changes
  const DIV_RATE = 100;          // 1 divine = X chaos — update with actual rate

  const BASES = [
    { name: 'Pronged Spear',  brackets: [{label:'<81',min:null,max:80},{label:'81',min:81,max:81},{label:'82+',min:82,max:null}] },
    { name: 'Guardian Spear', brackets: [{label:'<81',min:null,max:80},{label:'81',min:81,max:81},{label:'82+',min:82,max:null}] },
    { name: 'Stalking Spear', brackets: [{label:'<81',min:null,max:80},{label:'81',min:81,max:81},{label:'82+',min:82,max:null}] },
    { name: 'Spiked Spear',   brackets: [{label:'<81',min:null,max:80},{label:'81',min:81,max:81},{label:'82+',min:82,max:null}] },
    { name: 'Akoyan Spear',   brackets: [{label:'<81',min:null,max:80},{label:'81',min:81,max:81},{label:'82+',min:82,max:null}] },
    { name: 'Flying Spear',   brackets: [{label:'<81',min:null,max:80},{label:'81',min:81,max:81},{label:'82+',min:82,max:null}] },
    { name: 'Grand Spear',    brackets: [{label:'<81',min:null,max:80},{label:'81',min:81,max:81},{label:'82+',min:82,max:null}] }
  ];

  const delay = ms => new Promise(r => setTimeout(r, ms));
  const median = arr => {
    if (!arr.length) return null;
    const s = [...arr].sort((a,b) => a-b);
    const m = Math.floor(s.length/2);
    return s.length % 2 ? s[m] : (s[m-1]+s[m])/2;
  };
  const mean = arr => arr.length ? arr.reduce((a,b)=>a+b,0)/arr.length : null;

  const toDiv = (price, rate) => {
    if (!price) return null;
    if (price.currency === 'divine') return price.amount;
    if (price.currency === 'chaos')  return price.amount / rate;
    return null;
  };

  async function search(baseName, ilvlMin, ilvlMax) {
    const ilvlFilter = {};
    if (ilvlMin != null) ilvlFilter.min = ilvlMin;
    if (ilvlMax != null) ilvlFilter.max = ilvlMax;
    const body = {
      query: {
        status: { option: 'securable' },
        type: baseName,
        filters: {
          misc_filters: {
            filters: Object.keys(ilvlFilter).length ? { ilvl: ilvlFilter } : {}
          }
        },
        stats: [{ type: 'and', filters: [], disabled: false }]
      },
      sort: { price: 'asc' }
    };
    const r = await fetch(`https://www.pathofexile.com/api/trade2/search/poe2/${LEAGUE}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'X-Requested-With': 'XMLHttpRequest' },
      credentials: 'include',
      body: JSON.stringify(body)
    });
    if (!r.ok) return { error: `HTTP ${r.status}`, result: [], total: 0 };
    return r.json();
  }

  async function fetchBatch(ids, searchId) {
    const r = await fetch(
      `https://www.pathofexile.com/api/trade2/fetch/${ids.join(',')}?query=${searchId}&realm=poe2`,
      { credentials: 'include' }
    );
    if (!r.ok) return { result: [] };
    return r.json();
  }

  const results = {};
  for (const base of BASES) {
    results[base.name] = {};
    for (const bracket of base.brackets) {
      await delay(1200);
      const s = await search(base.name, bracket.min, bracket.max);
      if (!s.result || !s.id) {
        results[base.name][bracket.label] = { median10: null, median100: null, mean100: null, count: 0, error: s.error };
        continue;
      }
      const ids   = s.result.slice(0, 100);
      const total = s.total || 0;
      const prices = [];
      for (let i = 0; i < ids.length; i += 10) {
        if (i > 0) await delay(1200);
        const batch = await fetchBatch(ids.slice(i, i+10), s.id);
        for (const item of (batch.result || [])) {
          const p = toDiv(item?.listing?.price, DIV_RATE);
          if (p !== null) prices.push(p);
        }
      }
      const top10 = prices.slice(0, 10);
      results[base.name][bracket.label] = {
        min:       prices.length ? prices[0] : null,
        median10:  median(top10),
        mean10:    mean(top10),
        median100: median(prices),
        mean100:   mean(prices),
        count:     total
      };
    }
  }
  return { timestamp: new Date().toISOString(), divineRate: DIV_RATE, results };
});
```

## After Fetch — Updating prices.html
Replace the content of `<script id="prices-data" type="application/json">` in prices.html.

  → set median10, median100, mean100, count
- Set `lastUpdated` to the returned timestamp
- Set `divineRateChaos` to the used DIV_RATE
